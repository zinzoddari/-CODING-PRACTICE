큐는, 먼저 들어온 데이터가 먼저 나가는 FIRST IN FIRST OUT 구조이다.

```java
public class Queue {
    public static final int QUEUE_SIZE = 1000;
    public static final String[] array = new String[QUEUE_SIZE];
    public static int size = 0;
    public static int front = -1;

    //offer, peek, roll, isEmpty, size
    public void offer(String key) {
        array[++front] = key;
        size++;
    }

    public String peek() {
        return array[front];
    }

    public String roll() {
        if(isEmpty()) return null;
        String result = array[front];
        array[front] = null;
        size--;
        return result;
    }

    public boolean isEmpty() {
        return size > 0 ? false : true;
    }

    public int size() {
        return size;
    }
}
```

- 테스트코드
```java
import static org.junit.jupiter.api.Assertions.*;

class QueueTest {
    @Test
    void offer() {
        Queue queue = new Queue();
        queue.offer("zz");
        assertEquals(1, queue.size());
        assertEquals("zz", queue.peek());
    }


    @ParameterizedTest
    @NullSource
    void roll_null(String value) {
        Queue queue = new Queue();
        assertEquals(value, queue.roll());
    }

    @Test
    void roll() {
        String param = "zz";
        Queue queue = new Queue();
        queue.offer(param);

        String result = queue.roll();

        assertEquals(param, result);
    }

    @Test
    void size() {
        Queue queue = new Queue();
        queue.offer("zz");

        assertEquals(1, queue.size());
    }
}
```
